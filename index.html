function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet(); var type = data.type; var tab; var rowData; if (type === 'purchase') { tab = getOrCreateSheet(sheet, 'Purchases', ['Timestamp','Item Name','Item Type','Purchase Price (AUD)','Date Purchased','Platform','Invoice Status','Notes','File Attached']); rowData = [new Date(), data.itemName||'', data.itemType||'', data.price||'', data.date||'', data.platform||'', data.invoice||'', data.notes||'', data.file ? 'Yes - emailed to Erika' : 'No']; } else if (type === 'sale') { tab = getOrCreateSheet(sheet, 'Sales', ['Timestamp','Item Sold','Sale Price (AUD)','Date Sold','Platform','Buyer Name','Docs Status','Notes','File Attached']); rowData = [new Date(), data.itemName||'', data.price||'', data.date||'', data.platform||'', data.buyer||'', data.docs||'', data.notes||'', data.file ? 'Yes - emailed to Erika' : 'No']; } else if (type === 'grading') { tab = getOrCreateSheet(sheet, 'Grading', ['Timestamp','Item(s)','Grading Company','Date Sent','Submission Cost (AUD)','Status','Grade Result','Notes','File Attached']); rowData = [new Date(), data.itemName||'', data.company||'', data.date||'', data.cost||'', data.status||'', data.grade||'', data.notes||'', data.file ? 'Yes - emailed to Erika' : 'No']; } tab.appendRow(rowData); // Always send email notification notifyErika(data, type); return ContentService .createTextOutput(JSON.stringify({ result: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ result: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function notifyErika(data, type) { var recipientEmail = 'mendozaerikac9@gmail.com'; var subject = '[BV Capital] New ' + cap(type) + ' - ' + (data.itemName || 'Unknown Item'); var body = 'Hi Erika,\n\nBen has submitted a new ' + type + '.\n\n'; if (type === 'purchase') { body += 'Item: ' + (data.itemName || 'N/A') + '\n'; body += 'Item Type: ' + (data.itemType || 'N/A') + '\n'; body += 'Price: AUD $' + (data.price || 'N/A') + '\n'; body += 'Date: ' + (data.date || 'N/A') + '\n'; body += 'Platform: ' + (data.platform || 'N/A') + '\n'; body += 'Invoice Status: ' + (data.invoice || 'N/A') + '\n'; } else if (type === 'sale') { body += 'Item Sold: ' + (data.itemName || 'N/A') + '\n'; body += 'Sale Price: AUD $' + (data.price || 'N/A') + '\n'; body += 'Date: ' + (data.date || 'N/A') + '\n'; body += 'Platform: ' + (data.platform || 'N/A') + '\n'; body += 'Buyer: ' + (data.buyer || 'N/A') + '\n'; body += 'Docs: ' + (data.docs || 'N/A') + '\n'; } else if (type === 'grading') { body += 'Item(s): ' + (data.itemName || 'N/A') + '\n'; body += 'Grading Company: ' + (data.company || 'N/A') + '\n'; body += 'Date Sent: ' + (data.date || 'N/A') + '\n'; body += 'Cost: AUD $' + (data.cost || 'N/A') + '\n'; body += 'Status: ' + (data.status || 'N/A') + '\n'; body += 'Grade Result: ' + (data.grade || 'N/A') + '\n'; } if (data.notes) body += 'Notes: ' + data.notes + '\n'; body += '\nThis has been logged in the BV Capital Submissions Google Sheet.\n\nRegards,\nBV Capital Collectibles Portal'; // Attach file if present, otherwise just send plain email if (data.file && data.file.data) { var fileBytes = Utilities.base64Decode(data.file.data); var blob = Utilities.newBlob(fileBytes, data.file.type, data.file.name); GmailApp.sendEmail(recipientEmail, subject, body, { attachments: [blob] }); } else { GmailApp.sendEmail(recipientEmail, subject, body); } } function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet(); var result = { purchases: [], sales: [], grading: [] }; var purchaseTab = sheet.getSheetByName('Purchases'); if (purchaseTab && purchaseTab.getLastRow() > 1) { var pData = purchaseTab.getRange(2, 1, purchaseTab.getLastRow() - 1, 9).getValues(); result.purchases = pData.map(function(r) { return { timestamp: r[0], itemName: r[1], itemType: r[2], price: r[3], date: r[4], platform: r[5], invoice: r[6], notes: r[7], file: r[8] }; }).reverse(); } var salesTab = sheet.getSheetByName('Sales'); if (salesTab && salesTab.getLastRow() > 1) { var sData = salesTab.getRange(2, 1, salesTab.getLastRow() - 1, 9).getValues(); result.sales = sData.map(function(r) { return { timestamp: r[0], itemName: r[1], price: r[2], date: r[3], platform: r[4], buyer: r[5], docs: r[6], notes: r[7], file: r[8] }; }).reverse(); } var gradingTab = sheet.getSheetByName('Grading'); if (gradingTab && gradingTab.getLastRow() > 1) { var gData = gradingTab.getRange(2, 1, gradingTab.getLastRow() - 1, 9).getValues(); result.grading = gData.map(function(r) { return { timestamp: r[0], itemName: r[1], company: r[2], date: r[3], cost: r[4], status: r[5], grade: r[6], notes: r[7], file: r[8] }; }).reverse(); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function getOrCreateSheet(spreadsheet, name, headers) { var tab = spreadsheet.getSheetByName(name); if (!tab) { tab = spreadsheet.insertSheet(name); tab.appendRow(headers); tab.getRange(1, 1, 1, headers.length).setFontWeight('bold'); tab.setFrozenRows(1); } return tab; } function cap(s) { return s.charAt(0).toUpperCase() + s.slice(1); }
